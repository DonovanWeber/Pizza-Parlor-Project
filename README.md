const myPizza = ("large", ["Pepperoni", "Pineapple"])
Expected Output: Pizza; {size: "large", toppings:["Pepperoni", "Pineapple"]}

const myPizza = ("large", ["pepperoni"], 0)
myPizza.priceOfSize();
